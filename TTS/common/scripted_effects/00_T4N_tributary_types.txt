non_permanent = {
	make_custom_tributary = {
		TRIBUTARY = $TRIBUTARY$
		SUZERAIN = $SUZERAIN$
	}

	$TRIBUTARY$ = {
		set_variable = {
			name = possible_permafication_year
            value = { value = current_year add = 15 } 
        }
	}
}

	}

	$TRIBUTARY$ = {

            name = possible_vasalisation_year
